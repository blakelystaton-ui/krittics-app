4. interests.component.ts – Save to Firestore + matchmaking prep
ts// src/app/interests/interests.component.ts
import { Component, OnInit } from '@angular/core';
import { environment } from '../../environments/environment';
import { UserService } from '../shared/services/user.service';
import { AlertController } from '@ionic/angular';

@Component({
  selector: 'app-interests',
  templateUrl: './interests.component.html',
  styleUrls: ['./interests.component.scss'],
})
export class InterestsComponent implements OnInit {
  hobbies = environment.hobbies;
  selected: Record<string, boolean> = {};

  constructor(
    private user: UserService,
    private alertCtrl: AlertController
  ) {}

  ngOnInit() {
    this.user.getHobbies().subscribe(data => {
      this.selected = data || {};
    });
  }

  toggle(hobby: string) {
    this.selected[hobby] = !this.selected[hobby];
  }

  get hasSelection() {
    return Object.values(this.selected).some(v => v);
  }

  async save() {
    if (!this.hasSelection) return;

    await this.user.saveHobbies(this.selected);

    const alert = await this.alertCtrl.create({
      header: 'Ready for Krossfire!',
      message: 'Your hobbies are saved. Jump into quick match or invite friends.',
      buttons: ['Let’s Go']
    });
    await alert.present();
  }
}

5. user.service.ts – Store hobbies + matchmaking helpers
ts// src/app/shared/services/user.service.ts
import { Injectable } from '@angular/core';
import { AngularFirestore } from '@angular/fire/compat/firestore';
import { AuthService } from './auth.service';
import { switchMap, map } from 'rxjs/operators';
import { of } from 'rxjs';

@Injectable({ providedIn: 'root' })
export class UserService {
  private uid$ = this.auth.user$.pipe(map(u => u?.uid));

  constructor(private afs: AngularFirestore, private auth: AuthService) {}

  // Get user's saved hobbies
  getHobbies() {
    return this.uid$.pipe(
      switchMap(uid => uid
        ? this.afs.doc<Record<string, boolean>>(`users/${uid}`).valueChanges()
        : of({})
      )
    );
  }

  // Save hobbies
  saveHobbies(selected: Record<string, boolean>) {
    return this.uid$.pipe(
      switchMap(uid => uid
        ? this.afs.doc(`users/${uid}`).set({ hobbies: selected }, { merge: true })
        : of(null)
      )
    ).toPromise();
  }

  // Krossfire: Find random opponent with ≥1 shared hobby
  findRandomOpponent() {
    return this.uid$.pipe(
      switchMap(uid => uid
        ? this.afs.collection('users', ref => ref.where('hobbies', '!=', {}))
            .valueChanges({ idField: 'uid' })
            .pipe(map(users => users.filter(u => u.uid !== uid)))
        : of([])
      )
    );
  }

  // Send friend request
  sendFriendRequest(targetUid: string) {
    return this.uid$.pipe(
      switchMap(uid => uid
        ? this.afs.doc(`users/${targetUid}/friendRequests/${uid}`).set({
            from: uid,
            timestamp: new Date()
          })
        : of(null)
      )
    ).toPromise();
  }
}

6. ads.service.ts – Target AdSense with hobbies
ts// src/app/shared/services/ads.service.ts
import { Injectable } from '@angular/core';
import { UserService } from './user.service';

@Injectable({ providedIn: 'root' })
export class AdsService {
  constructor(private user: UserService) {}

  loadAd(slotId: string) {
    this.user.getHobbies().subscribe(hobbies => {
      const keywords = Object.keys(hobbies).filter(k => hobbies[k]);
      const kwString = keywords.length ? keywords.join(',') : 'lifestyle,fun';

      (window.adsbygoogle = window.adsbygoogle || []).push({
        google_ad_client: 'ca-pub-YOUR_ID',
        enable_page_level_ads: true,
        overrides: { keywords: kwString }
      });
    });
  }
}
