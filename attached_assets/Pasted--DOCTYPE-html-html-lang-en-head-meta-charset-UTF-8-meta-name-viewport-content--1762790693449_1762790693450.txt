  <script>
    const FADE_DURATION = 1200;
    const AUTO_PLAY_INTERVAL = 4000;

    const slides = document.querySelectorAll('.slide');
    const dots   = document.querySelectorAll('.dot');
    const prevBtn = document.querySelector('.prev');
    const nextBtn = document.querySelector('.next');

    let currentIndex = 0;
    let timer = null;

    function goTo(idx) {
      idx = (idx + slides.length) % slides.length;
      document.querySelector('.slide.active')?.classList.remove('active');
      document.querySelector('.dot.active')?.classList.remove('active');
      slides[idx].classList.add('active');
      dots[idx].classList.add('active');
      currentIndex = idx;
    }

    function next() { goTo(currentIndex + 1); }
    function prev() { goTo(currentIndex - 1); }

    prevBtn.onclick = () => { stopAuto(); prev(); startAuto(); };
    nextBtn.onclick = () => { stopAuto(); next(); startAuto(); };

    dots.forEach(d => d.onclick = () => {
      const i = +d.dataset.index;
      stopAuto(); goTo(i); startAuto();
    });

    function startAuto() {
      if (!AUTO_PLAY_INTERVAL) return;
      stopAuto();
      timer = setInterval(next, AUTO_PLAY_INTERVAL);
    }
    function stopAuto() { if (timer) clearInterval(timer); }

    goTo(0);
    startAuto();
  </script>
